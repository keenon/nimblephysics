279 //
280 // Dev Note: We achieve "multiple inheritance" without the diamond of death
281 // issue by specifying detail::EmbeddedStateAspect as the base class of
282 // detail::EmbeddedPropertiesAspect. This allows their implementations to stack
283 // on top of each other without the conflict that would arise from both of them
284 // inheriting from common::Aspect.
285 template <class CompositeT, typename StateDataT, typename PropertiesDataT>
286 class EmbeddedStateAndPropertiesAspect :
287  public detail::EmbeddedPropertiesAspect<
288  detail::EmbeddedStateAspect<
289  CompositeTrackingAspect<CompositeT>,
290  EmbeddedStateAndPropertiesAspect<CompositeT, StateDataT, PropertiesDataT>,
291  StateDataT>,
292  EmbeddedStateAndPropertiesAspect<CompositeT, StateDataT, PropertiesDataT>,
293  PropertiesDataT>
294 {
295 public:
296 
297  using Derived = EmbeddedStateAndPropertiesAspect<CompositeT, StateDataT, PropertiesDataT>;
298 
299  using AspectStateImpl = detail::EmbeddedStateAspect<
300  CompositeTrackingAspect<CompositeT>, Derived, StateDataT>;
301 
302  using AspectPropertiesImpl = detail::EmbeddedPropertiesAspect<
303  AspectStateImpl, Derived, PropertiesDataT>;
304 
305  using AspectImpl = Derived;
306 
307  using State = typename AspectStateImpl::State;
308  using StateData = typename AspectStateImpl::StateData;
309 
310  using Properties = typename AspectPropertiesImpl::Properties;
311  using PropertiesData = typename AspectPropertiesImpl::PropertiesData;
312 
313  using CompositeType = CompositeT;
314 
315  EmbeddedStateAndPropertiesAspect(
316  const EmbeddedStateAndPropertiesAspect&) = delete;
317 
318  virtual ~EmbeddedStateAndPropertiesAspect() = default;
319 
321  EmbeddedStateAndPropertiesAspect()
322  : AspectPropertiesImpl()
323  {
324  // Do nothing
325  }
326 
328  EmbeddedStateAndPropertiesAspect(
329  const StateData& state)
330  : AspectPropertiesImpl(state)
331  {
332  // Do nothing
333  }
334 
336  EmbeddedStateAndPropertiesAspect(
337  const PropertiesData& properties)
338  : AspectPropertiesImpl(properties)
339  {
340  // Do nothing
341  }
342 
344  EmbeddedStateAndPropertiesAspect(
345  const StateData& state,
346  const PropertiesData& properties)
347  : AspectPropertiesImpl(properties, state)
348  {
349  // Do nothing
350  }
351 
353  EmbeddedStateAndPropertiesAspect(
354  const PropertiesData& properties,
355  const StateData& state)
356  : AspectPropertiesImpl(properties, state)
357  {
358  // Do nothing
359  }
360 
361  // Documentation inherited
362  std::unique_ptr<Aspect> cloneAspect() const override
363  {
364  return make_unique<Derived>(this->getState(), this->getProperties());
365  }
366 
367 };
